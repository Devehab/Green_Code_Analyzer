// Inefficient JavaScript code example
function findDuplicates(array) {
  const duplicates = [];
  
  // Inefficient nested loop approach
  for (let i = 0; i < array.length; i++) {
    for (let j = 0; j < array.length; j++) {
      // Skip comparing the same index
      if (i === j) continue;
      
      // Check for duplicate
      if (array[i] === array[j] && !duplicates.includes(array[i])) {
        duplicates.push(array[i]);
      }
    }
  }
  
  return duplicates;
}

import java.util.ArrayList;
import java.util.List;
import java.util.Random;

public class InefficiencyExample {
    
    public static List findDuplicates(List array) {
        List duplicates = new ArrayList<>();
        
        // Inefficient nested loop approach
        for (int i = 0; i < array.size(); i++) {
            for (int j = 0; j < array.size(); j++) {
                // Skip comparing the same index
                if (i == j) continue;
                
                // Check for duplicate
                if (array.get(i).equals(array.get(j)) && !duplicates.contains(array.get(i))) {
                    duplicates.add(array.get(i));
                }
            }
        }
        
        return duplicates;
    }
    
    // Generate a large array for testing
    public static List generateLargeArray(int size) {
        List result = new ArrayList<>();
        Random random = new Random();
        
        for (int i = 0; i < size; i++) {
            result.add(random.nextInt(100));
        }
        
        return result;
    }
    
    // Process the array multiple times unnecessarily
    public static void processData() {
        List data = generateLargeArray(1000);
        
        // Unnecessary multiple iterations
        int sum = 0;
        for (int i = 0; i < data.size(); i++) {
            sum += data.get(i);
        }
        
        int max = Integer.MIN_VALUE;
        for (int i = 0; i < data.size(); i++) {
            if (data.get(i) > max) {
                max = data.get(i);
            }
        }
        
        int min = Integer.MAX_VALUE;
        for (int i = 0; i < data.size(); i++) {
            if (data.get(i) < min) {
                min = data.get(i);
            }
        }
        
        // Find duplicates using inefficient function
        List duplicates = findDuplicates(data);
        
        // Create a new object for each result
        Result result = new Result(sum, max, min, duplicates);
    }
    
    public static void main(String[] args) {
        while (true) {
            processData();
            try {
                Thread.sleep(1000);
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }
    }
    
    static class Result {
        int sum;
        int max;
        int min;
        List duplicates;
        
        public Result(int sum, int max, int min, List duplicates) {
            this.sum = sum;
            this.max = max;
            this.min = min;
            this.duplicates = duplicates;
        }
    }
}
